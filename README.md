# bs-amap

## Usage
```
npm install @xinguang-data/bs-amap
```

```
import bsAmap from '@xinguang-data/bs-amap'
import vue from 'vue'
vue.use( bsAmap );
```

```
<bs-amap
    :region_id=""
    :mapConfig=""
    :polygons=""
    :basePolygons=""
    :polygonStyle=""
    :polygonDyeing=""
    :pointData=""
    :satelliteIsShow=""
>

</bs-amap>
```
